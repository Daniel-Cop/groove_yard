RegistrationConfirmationSubscriber
    <?php

namespace App\EventSubscriber;

use App\Entity\User;
use App\Mail\UserRegisteredConfirmation;
use Doctrine\Bundle\DoctrineBundle\Attribute\AsDoctrineListener;
use Doctrine\ORM\Event\PrePersistEventArgs;
use Doctrine\ORM\Events;

// At user registration send a confirmation email
#[AsDoctrineListener(Events::prePersist)]
class RegistrationConfirmationSubscriber
{
    public function __construct(
        private UserRegisteredConfirmation $mailer
    ) {
    }

    public function prePersist(PrePersistEventArgs $event): void
    {
        $entity = $event->getObject();

        if (!$entity instanceof User) {
            return;
        }

        $this->mailer->sendRegistrationConfirmation($entity);
    }
}
